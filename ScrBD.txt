Use SS_Bibliotecario;
CREATE TABLE Libros(IdLibro varchar(50) PRIMARY KEY, Nombre varchar(50), Autor  varchar(50), editorial  varchar(50), FechaIngreso date, Dispon boolean);
CREATE TABLE Login(Correo varchar (50) PRIMARY KEY, Contraseña varchar(50), IdSuscrip varchar(50), CONSTRAINT FKLog_Susc FOREIGN KEY (IdSuscrip) REFERENCES Suscrip(IdSuscrip));
CREATE TABLE SolLibros(IdSL varchar (50) PRIMARY KEY, Correo varchar (50), IdLibro varchar (50), CONSTRAINT FKSLL_Log FOREIGN KEY Correo REFERENCES Login(Correo), CONSTRAINT FKSLL_Lib FOREIGN KEY IdLibro REFERENCES Libros(IdLibro));
CREATE TABLE Registro(IdRegistro varchar (50) PRIMARY KEY,Correo varchar(50), Contraseña varchar(50), Nombre varchar(50), Apellido varchar(50), FeNa date, Sexo boolean, Telefono int, CONSTRAINT FKReg_Log FOREIGN KEY (Correo) REFERENCES Login(Correo));
CREATE TABLE Suscrip(IdSuscrip varchar (50) PRIMARY KEY, Valor int, Benef varchar(50), FeRenov date);
CREATE TABLE ListCategor(IdLC varchar (50) PRIMARY KEY, IdLibro varchar (50), IdCateg varchar (50), CONSTRAINT FKLiCa_Lib FOREIGN KEY (IdLibro) REFERENCES Libros(IdLibro), CONSTRAINT FKSLL_Cat FOREIGN KEY (IdCateg) REFERENCES Categoria(IdCateg));
CREATE TABLE Categoria(IdCateg varchar (50) PRIMARY KEY, Genero varchar (50));
